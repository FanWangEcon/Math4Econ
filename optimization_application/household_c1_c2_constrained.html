% Parameters
beta = 0.90;
z1 = 1;
z2 = 2;
r = 0.05;
% Write down the objective function, we will define it as a function handle, negative utility for minimization
U_neg = @(x) -1*(log(x(2)) + beta*log(x(1)));
% Constraint derived above
A = [1,1+r;0,-1;-1,0];
