This is a function:
figure();
x = 0:pi/100:2*pi;
y = sin(x);
plot(x,y);
grid on;
This is NOT a function:
figure();
x = 1; y=1; r=1;
th = 0:pi/50:2*pi;
xunit = r * cos(th) + x;
yunit = r * sin(th) + y;
h = plot(xunit, yunit);
grid on;
